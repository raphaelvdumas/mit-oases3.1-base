% MIT definition -->
      \centerline{\psfig{figure=#1,height=#2cm}}

\newcommand{\figmittwo}[6]{
    \begin{figure}
% MIT definition -->
       \vbox{\centerline{\psfig{figure=#1,height=#2cm}}}
       \vbox{\centerline{\psfig{figure=#3,height=#4cm}}}
       \caption{#5} \label{#6}
    \end{figure}          }
